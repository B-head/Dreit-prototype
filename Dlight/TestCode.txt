import basic;

stdout := "Hallo World"; // おはよう。

class FizzBuzz
{
	function output(num1:Integer32, num2:Integer32)
	{
		for(0..100 :: i)
		{ 
			var c1 := i % num1 = 0;
			var c2 := i % num2 = 0;
			if(c1 && c2) stdout ;= "FizzBuzz";
			else if(c1) stdout ;= "Fizz";
			else if(c2) stdout ;= "Buzz";
			else stdout := i;
		}
	}
}
