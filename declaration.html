ImportDeclaration:

AliasDeclaration:
	"alias", AliasList, PrefixExpression

AliasList:
	Identifier (,",", Identifier)*

alias after before

変数

VariantDeclaration:
	(AttributeSignature,)? VariantPrefix, Identifier (,":", PrefixExpression)?

VariantPrefix:
	"var"
	"let"
	"const"

var a
let b
const c

列挙型未実装

EnumDeclaration:
	(AttributeSignature,)? "enum" (,Identifier)? (,":", PrefixExpression)? (,EnumContext)?

EnumContext:
	("::",)? ";"
	"::", EnumField, (";")?
	("::",)? "{", EnumList, "}"

EnumList:
	;? EnumField (; EnumField)* ;?
	;

EnumField:
	EnumFieldPart, ":=", NakedRangeLiteral
	NakedRangeLiteral, "=:", EnumFieldPart
	ArgumentPart

EnumFieldPart:
	(AttributeSignature,)? Identifier

enum : Int32 {
	foo := 0
	bar
	baz
	qux := 99
}

属性

AttributeSignature:
	StartAttribute, FollowAttribute*

StartAttribute:
	("@",)? EmbedAttribute
	"@", NakedRangeLiteral

FollowAttribute:
	",", Attribute
	StartAttribute

Attribute:
	StartAttribute
	NakedRangeLiteral

EmbedAttribute:
	"static"
	"override"
	"final"
	"variadic"
	"private"
	"protected"
	"internal"
	"public"

public static var a
